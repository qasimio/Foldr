# ── Core (required on all platforms) ─────────────────────────────────────────

# File system event watching — used by `foldr watch`
# OS-native backends, zero CPU when idle:
#   Linux   → inotify
#   macOS   → FSEvents / kqueue
#   Windows → ReadDirectoryChangesW
watchdog>=4.0.0

# Table formatting for history / dedup / watches output
# FOLDR falls back to plain text if missing, but tables look better with it
tabulate>=0.9.0

# TOML config file support for Python 3.10
# Python 3.11+ has tomllib built-in — tomli is skipped automatically
tomli>=2.0.1 ; python_version < "3.11"


# ── Windows only (installed automatically by pip on Windows) ──────────────────

# ANSI colour support in cmd.exe and older PowerShell
# Win10 build 14931+ is auto-enabled via ctypes — colorama is a safety net
colorama>=0.4.6 ; sys_platform == "win32"
