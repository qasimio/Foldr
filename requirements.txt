# ── Required on all platforms ─────────────────────────────────────────────────

# File system event watching — used by `foldr watch`
# OS-native, zero CPU when idle:
#   Linux   → inotify
#   macOS   → FSEvents / kqueue
#   Windows → ReadDirectoryChangesW
watchdog>=4.0.0

# Table formatting for history / watches / dedup output
# Graceful plain-text fallback if missing, but tables look much better with it
tabulate>=0.9.0

# TOML config support for Python 3.10
# Python 3.11+ includes tomllib in the standard library — tomli is skipped automatically
tomli>=2.0.1 ; python_version < "3.11"

# ── Windows only ──────────────────────────────────────────────────────────────

# ANSI colour in cmd.exe and older PowerShell
# Windows 10 build 14931+ is auto-enabled via ctypes — colorama is a safety net
colorama>=0.4.6 ; sys_platform == "win32"
